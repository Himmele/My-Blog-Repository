#include "pmode.h"

void main (void)
	{
    // Debug-Register 0 auf Funktion print setzen!
    set_drx (0, ACCESS_READ, ACCESS_BYTE, &print);

    print (10, 10, "Hallo Welt vom Protected Mode aus BC 4.0 !!!", 0x0a);
	}
